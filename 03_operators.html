Operators

       
        let a = 10,
            b = 4,
            c = 15;
        let d = '5'
            e = 'my name'
            f = 'is max';

        // Addition
        console.group('Addition');
            console.log(a + b); // 14
            console.log(a + c); // 25
            console.log(e + ' ' + f);   // my name is max
            console.log(a + d); // 105
            console.log(e + ' is batman');  // my name is batman
        console.groupEnd();

        // Substraction
        console.group('Substraction');
            console.log(a - b); // 6
            console.log(a - c); // -5
            console.log(a - d); // 5
        console.groupEnd();

        // Multiplication
        console.group('Multiplication');
            console.log(a * 3); // 30
            console.log(a * b); // 40
            console.log(b * c); // 60
        console.groupEnd();

        // Division
        console.group('Division');
            console.log(a / 2); // 5
            console.log(a / 3); // 3.3333333
        console.groupEnd();

        // Incrementing
        console.group('Incrementing');
            console.log(a + 1); // 11
            console.log(a++);   // 10
            console.log(a); // 11
            console.log(++a); // 12
            console.log(b += 2);    // 6
        console.groupEnd();

        // Decrementing
        console.group('Decrementing')
            console.log(c - 1); // 14
            console.log(--c);   // 14
            console.log(b -= 2);    // 4
        console.groupEnd();
